# step1: 重命名文件

由于文件的名字里有数字 1，2，…10，11…, 后续需要根据这个顺序拼接，但是这个数字不好提取，有一种办法就是将字符串特定位置（这里只有两位数，索引为 [-6,-4]）的字符取出来，那么就必须规定数字字符占的位一致，所以需要将个位数字前面加一个 0, 这样这些数字都可以通过【转换为列表，取索引 [-6,-4] 的字符，转为 int，排序】这个过程提取出来了。

import os
def rename():
    for j in [i for i in os.listdir() if os.path.isdir(i)]:
        file_list = os.listdir(j)
        path = os.path.join(os.getcwd(),j)
        for file in file_list:
            # file = file.replace(' ','') # 去除里面的空格 
            li = list(file)
            if li[-6]=='-':
                li.insert(-5,'0')
                new_name = "".join(li)
                os.rename(os.path.join(path,file),os.path.join(path,new_name))

# step2: 去除文件名中的空格

for j in [i for i in os.listdir() if os.path.isdir(i)]:
    file_list = os.listdir(j)
    path = os.path.join(os.getcwd(),j)
    for file in file_list:
        new_name = file.replace('：','-') # 去除里面的空格，或者替换掉不好的字符
        os.rename(os.path.join(path,file),os.path.join(path,new_name))

# step3: 写文件

由于文件名字中的数字统一了占位，在读取为列表时，已经自动排好序了，所以按照遍历的顺序写入就可以了。

def write():
    for j in [i for i in os.listdir() if os.path.isdir(i)]:
        f = open(j+'.md','w')
        f.write('---\n')
        f.write('title: '+j+'\n')
        f.write('categories: '+'python\n')
        f.write('tags: '+'python\n')
        f.write('---\n')
        for file in os.listdir(j):
            f.write('![](/images/'+file+')\n')
        f.close()
